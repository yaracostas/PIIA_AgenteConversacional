  async function startConversation() {
    if (conversationActive) return;

    conversationActive = true;
    setStatus("Conectando con el servidor...", true);
    removeThinkingMessage();

    try {
      socket = new WebSocket(WS_URL);

      socket.onopen = async () => {
        console.log("WebSocket conectado:", WS_URL);
        setStatus("Conectado. Activando micrófono...", true);

        await startMicrophoneStreaming();
      };

      socket.onmessage = async (event) => {
        await handleServerMessage(event);
      };

      socket.onerror = (error) => {
        console.error("Error WebSocket:", error);
        setStatus("Error de conexión", false);
        addAgentMessage(
          "No he podido conectar con el servidor. Comprueba que Docker esté levantado y que el servidor esté escuchando en el puerto 8770.",
          "idle"
        );
        stopConversationLocally();
      };

      socket.onclose = () => {
        console.log("WebSocket cerrado");
        stopConversationLocally();
        setStatus("Conversación detenida", false);
      };

    } catch (error) {
      console.error(error);
      setStatus("Error al iniciar", false);
      addAgentMessage(
        "No se pudo iniciar la conversación. Revisa la conexión con el servidor y los permisos del navegador.",
        "idle"
      );
      stopConversationLocally();
    }
  }

  async function startMicrophoneStreaming() {
    try {
      audioStream = await navigator.mediaDevices.getUserMedia({
        audio: {
          echoCancellation: true,
          noiseSuppression: true,
          autoGainControl: true
        }
      });

      const mimeType = getSupportedMimeType();

      if (!mimeType) {
        addAgentMessage(
          "Tu navegador no soporta un formato de grabación compatible con MediaRecorder.",
          "idle"
        );
        endConversation();
        return;
      }

      mediaRecorder = new MediaRecorder(audioStream, { mimeType });

      mediaRecorder.ondataavailable = async (event) => {
        if (
          event.data &&
          event.data.size > 0 &&
          socket &&
          socket.readyState === WebSocket.OPEN
        ) {
          socket.send(event.data);
        }
      };

      mediaRecorder.onerror = (event) => {
        console.error("Error en MediaRecorder:", event);
        addAgentMessage("Ha ocurrido un error capturando el audio del micrófono.", "idle");
        endConversation();
      };

      /*
        Cada 250 ms se envía un fragmento de audio al backend.
        Si queréis menos latencia, podéis probar 100 o 150.
      */
      mediaRecorder.start(250);

      setStatus("Escuchando...", true);
      addAgentMessage("Te escucho. Puedes empezar a hablar.", "idle");

    } catch (error) {
      console.error(error);
      addAgentMessage(
        "No he podido acceder al micrófono. Revisa los permisos del navegador.",
        "idle"
      );
      endConversation();
    }
  }

  function getSupportedMimeType() {
    const types = [
      "audio/webm;codecs=opus",
      "audio/webm",
      "audio/ogg;codecs=opus",
      "audio/ogg"
    ];

    for (const type of types) {
      if (MediaRecorder.isTypeSupported(type)) {
        return type;
      }
    }

    return "";
  }

  async function handleServerMessage(event) {
    /*
      El servidor puede mandar:
      - JSON con eventos: {"type": "...", "text": "..."}
      - audio binario directamente
    */

    if (event.data instanceof Blob) {
      await playAssistantAudioFromBlob(event.data);
      return;
    }

    let data;

    try {
      data = JSON.parse(event.data);
    } catch (error) {
      console.warn("Mensaje recibido no JSON:", event.data);
      return;
    }

    console.log("Mensaje del servidor:", data);

    switch (data.type) {
      case "ready":
        setStatus("Servidor listo", true);
        break;

      case "listening":
        setStatus("Escuchando...", true);
        setAgentSpeaking(false);
        break;

      case "user_transcript":
      case "transcription":
      case "user_text":
        if (data.text) {
          addUserMessage(data.text);
        }
        break;

      case "assistant_thinking":
      case "thinking":
        setAgentSpeaking(false);
        addThinkingMessage();
        setStatus("TresPi está pensando...", true);
        break;

      case "assistant_started":
      case "tts_started":
      case "speaking_started":
        removeThinkingMessage();
        setAgentSpeaking(true);
        setStatus("TresPi está hablando...", true);
        break;

      case "assistant_text":
      case "llm_response":
      case "response_text":
        removeThinkingMessage();
        if (data.text) {
          addAgentMessage(data.text, "talking");
        }
        break;

      case "assistant_audio_base64":
      case "audio_base64":
        removeThinkingMessage();
        if (data.audio) {
          await playAssistantAudioBase64(data.audio);
        }
        break;

      case "assistant_audio_url":
      case "audio_url":
        removeThinkingMessage();
        if (data.url) {
          await playAssistantAudioUrl(data.url);
        }
        break;

      case "assistant_finished":
      case "tts_finished":
      case "speaking_finished":
        setAgentSpeaking(false);
        setStatus("Escuchando...", true);
        break;

      case "interrupted":
        stopAssistantAudio();
        setAgentSpeaking(false);
        removeThinkingMessage();
        setStatus("Interrupción detectada", true);
        break;

      case "error":
        removeThinkingMessage();
        setAgentSpeaking(false);
        addAgentMessage(data.message || "Ha ocurrido un error en el servidor.", "idle");
        setStatus("Error", false);
        break;

      default:
        console.warn("Tipo de mensaje no reconocido:", data);
        break;
    }
  }

  function endConversation() {
    if (!conversationActive) {
      setStatus("La conversación ya estaba detenida", false);
      return;
    }

    stopConversationLocally();
    addAgentMessage("Conversación finalizada. Hasta luego.", "idle");
    setStatus("Conversación detenida", false);
  }

  function stopConversationLocally() {
    conversationActive = false;

    removeThinkingMessage();
    stopAssistantAudio();
    setAgentSpeaking(false);

    if (mediaRecorder && mediaRecorder.state !== "inactive") {
      mediaRecorder.stop();
    }

    if (audioStream) {
      audioStream.getTracks().forEach(track => track.stop());
    }

    if (socket && socket.readyState === WebSocket.OPEN) {
      socket.close();
    }

    mediaRecorder = null;
    audioStream = null;
    socket = null;
  }

  async function playAssistantAudioUrl(audioUrl) {
    stopAssistantAudio();
    setAgentSpeaking(true);
    setStatus("TresPi está hablando...", true);

    assistantAudio = new Audio(audioUrl);

    assistantAudio.onended = () => {
      setAgentSpeaking(false);
      setStatus("Escuchando...", true);
    };

    assistantAudio.onerror = () => {
      setAgentSpeaking(false);
      addAgentMessage("No he podido reproducir el audio de la respuesta.", "idle");
    };

    try {
      await assistantAudio.play();
    } catch (error) {
      console.error(error);
      setAgentSpeaking(false);
      addAgentMessage("El navegador bloqueó la reproducción del audio.", "idle");
    }
  }

  async function playAssistantAudioBase64(base64Audio) {
    stopAssistantAudio();
    setAgentSpeaking(true);
    setStatus("TresPi está hablando...", true);

    assistantAudio = new Audio("data:audio/wav;base64," + base64Audio);

    assistantAudio.onended = () => {
      setAgentSpeaking(false);
      setStatus("Escuchando...", true);
    };

    assistantAudio.onerror = () => {
      setAgentSpeaking(false);
      addAgentMessage("No he podido reproducir el audio generado por el TTS.", "idle");
    };

    try {
      await assistantAudio.play();
    } catch (error) {
      console.error(error);
      setAgentSpeaking(false);
      addAgentMessage("El navegador bloqueó la reproducción del audio.", "idle");
    }
  }

  async function playAssistantAudioFromBlob(blob) {
    stopAssistantAudio();
    setAgentSpeaking(true);
    setStatus("TresPi está hablando...", true);

    const audioUrl = URL.createObjectURL(blob);
    assistantAudio = new Audio(audioUrl);

    assistantAudio.onended = () => {
      URL.revokeObjectURL(audioUrl);
      setAgentSpeaking(false);
      setStatus("Escuchando...", true);
    };

    assistantAudio.onerror = () => {
      URL.revokeObjectURL(audioUrl);
      setAgentSpeaking(false);
      addAgentMessage("No he podido reproducir el audio recibido.", "idle");
    };

    try {
      await assistantAudio.play();
    } catch (error) {
      console.error(error);
      setAgentSpeaking(false);
      addAgentMessage("El navegador bloqueó la reproducción del audio.", "idle");
    }
  }

  function stopAssistantAudio() {
    if (assistantAudio) {
      assistantAudio.pause();
      assistantAudio.currentTime = 0;
      assistantAudio = null;
    }
  }
